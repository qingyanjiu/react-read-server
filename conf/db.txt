1.user_info 用户信息
  create table user_info (user_id int(4) key not null auto_increment,user_name varchar(20) not null, 
  password varchar(100) not null,gender varchar(1),head varchar(3),regist_date varchar(20));


2.book_info 书籍信息
  CREATE TABLE `book_info` (                                                                                                                                                                                                                                 
  `id` int(11) NOT NULL AUTO_INCREMENT COMMENT 'id',                                                                                                                                                                                                                     
  `name` varchar(32) NOT NULL COMMENT '书名',                                                                                                                                                                                                                            
  `description` varchar(1000) DEFAULT NULL,                                                                                                                                                                                                                              
  `douban_id` varchar(20) DEFAULT NULL,                                                                                                                                                                                                                                  
  `image_url` varchar(100) DEFAULT NULL,                                                                                                                                                                                                                                 
  `plan_date` varchar(20) DEFAULT NULL,                                                                                                                                                                                                                                  
  `status` varchar(1) DEFAULT NULL,                                                                                                                                                                                                                                      
  `user_id` int(4) DEFAULT NULL,                                                                                                                                                                                                                                         
  `read_plan_month` varchar(20) DEFAULT NULL,                                                                                                                                                                                                                            
  PRIMARY KEY (`id`)                                                                                                                                                                                                                                                     
) ENGINE=InnoDB AUTO_INCREMENT=23 DEFAULT CHARSET=utf8 COMMENT='书籍表'


3.read_history 阅读历史
  CREATE TABLE `read_history` (                                                                                                                                                                                                                                 
  `id` int(11) NOT NULL AUTO_INCREMENT COMMENT 'id',                                                                                                                                                                                                                     
  `douban_id` varchar(32) NOT NULL COMMENT '豆瓣id',                                                                                                                                                                                                                            
  `user_id` varchar(100) DEFAULT NULL,
  `times` int(3) NOT NULL COMMENT '当前阅读次数',
  `read_time` varchar(5) DEFAULT NULL COMMENT '当前读的遍数',
  `start_date` varchar(20) DEFAULT NULL COMMENT '这一遍启读时间',                                                                                                                                                                                                                                  
  `end_date` varchar(20) DEFAULT NULL COMMENT '这一遍毕读时间',    
  `tag` varchar(1) DEFAULT '0' COMMENT '表示当前阅读状态0-启读 1-毕读 2-弃坑',
  PRIMARY KEY (`id`)                                                                                                                                                                                                                                                     
) ENGINE=InnoDB AUTO_INCREMENT=23 DEFAULT CHARSET=utf8 COMMENT='阅读历史表'



4.book_note 阅读笔记 
  CREATE TABLE `book_note` (                                                                                                                                                                                                                                 
  `id` int(11) NOT NULL AUTO_INCREMENT COMMENT 'id',                                                                                                                                                                                                                     
  `douban_id` varchar(32) NOT NULL COMMENT '豆瓣id',                                                                                                                                                                                                                            
  `user_id` varchar(100) DEFAULT NULL,      
  `page` varchar(5) DEFAULT NULL COMMENT '页码',
  `content` varchar(100) DEFAULT NULL COMMENT '文字内容',                                                                                                                                                                                                                                  
  `note` varchar(500) DEFAULT NULL COMMENT '笔记内容',   
  `note_date` varchar(20) DEFAULT NULL COMMENT '做笔记时间',
  PRIMARY KEY (`id`)                                                                                                                                                                                                                                                     
) ENGINE=InnoDB AUTO_INCREMENT=23 DEFAULT CHARSET=utf8 COMMENT='阅读笔记表'



5.book_tag 阅读书签 
  CREATE TABLE `book_tag` (                                                                                                                                                                                                                                 
  `id` int(11) NOT NULL AUTO_INCREMENT COMMENT 'id', 
  `douban_id` varchar(32) NOT NULL COMMENT '豆瓣id',
  `user_id` varchar(100) DEFAULT NULL,      
  `page` varchar(5) DEFAULT NULL COMMENT '页码',
  `tag_date` varchar(20) DEFAULT NULL COMMENT '书签时间',
  PRIMARY KEY (`id`)                                                                                                                                                                                                                                                     
) ENGINE=InnoDB AUTO_INCREMENT=23 DEFAULT CHARSET=utf8 COMMENT='阅读书签表'



6.book_comment 书籍评价 
  CREATE TABLE `book_comment` (                                                                                                                                                                                                                                 
  `id` int(11) NOT NULL AUTO_INCREMENT COMMENT 'id', 
  `douban_id` varchar(32) NOT NULL COMMENT '豆瓣id',
  `user_id` varchar(100) DEFAULT NULL,      
  `comment_content` varchar(1000) DEFAULT NULL COMMENT '评价内容',
  `comment_date` varchar(20) DEFAULT NULL COMMENT '评价时间',
  PRIMARY KEY (`id`)                                                                                                                                                                                                                                                     
) ENGINE=InnoDB AUTO_INCREMENT=23 DEFAULT CHARSET=utf8 COMMENT='书籍评价表'
